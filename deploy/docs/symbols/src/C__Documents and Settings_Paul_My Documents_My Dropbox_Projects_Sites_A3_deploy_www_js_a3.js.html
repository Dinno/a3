 85 /**
 86  * @class Represents a 3D Matrix <strong>[A3.M3]</strong>. Used much less than
 87  * the 4D matrix, but still has a special place in everyone's heart. Like the
 88  * awkward younger brother at the party.
 89  *
 90  * @author Paul Lewis
 91  */
 92 A3.Core.Math.Matrix3 = function() {
 93 	
 94 	this.m11 = 0; this.m12 = 0; this.m13 = 0;
 95 	this.m21 = 0; this.m22 = 0; this.m23 = 0;
 96 	this.m31 = 0; this.m32 = 0; this.m33 = 0;
 97 	
 98 };
 99 
100 A3.Core.Math.Matrix3.prototype = {
101 	
102 	/**
103 	 * Sets the values on the matrix
104 	 * 
105 	 * @param {Number} m11 Row 1, Column 1
106 	 * @param {Number} m12 Row 1, Column 2
107 	 * @param {Number} m13 Row 1, Column 3
108 	 * @param {Number} m21 Row 2, Column 1
109 	 * @param {Number} m22 Row 2, Column 2
110 	 * @param {Number} m23 Row 2, Column 3
111 	 * @param {Number} m31 Row 3, Column 1
112 	 * @param {Number} m32 Row 3, Column 2
113 	 * @param {Number} m33 Row 3, Column 3
114 	 */
115 	set: function(m11,m12,m13,
116 					m21,m22,m23,
117 					m31,m32,m33) {
118 	
119 		this.m11 = m11; this.m12 = m12; this.m13 = m13;
120 		this.m21 = m21; this.m22 = m22; this.m23 = m23;
121 		this.m31 = m31; this.m32 = m32; this.m33 = m33;
122 	},
123 	
124 	/**
125 	 * Turns the columns into rows, and the
126 	 * rows into columns.
127 	 */
128 	transpose: function() {
129 		
130 		var m12 = this.m12,
131         m13 = this.m13,
132         m21 = this.m21,
133         m23 = this.m23,
134         m31 = this.m31,
135         m32 = this.m32;
136 			
137 		this.m12 = m21; this.m13 = m31;
138 		this.m21 = m12; this.m23 = m32;
139 		this.m31 = m13; this.m32 = m23;
140 	},
141 	
142 	/**
143 	 * Calculates the determinant of the matrix. Primarily used for inverting the matrix
144 	 * 
145 	 * @see http://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/threeD/index.htm
146 	 */
147 	determinant: function() {
148 		
149 		return this.m11 * this.m22 * this.m33 +
150            this.m12 * this.m23 * this.m31 +
151            this.m13 * this.m21 * this.m32 -
152            this.m11 * this.m23 * this.m32 -
153            this.m12 * this.m21 * this.m33 -
154            this.m13 * this.m22 * this.m31;
155 	},
156 	
157 	/**
158 	 * Inverts the matrix
159 	 * 
160 	 * @throws An error if the matrix determinant is zero
161 	 */
162 	invert: function() {
163 		
164 		var m11 = this.m11, m12 = this.m12, m13 = this.m13,
165         m21 = this.m21, m22 = this.m22, m23 = this.m23,
166         m31 = this.m31, m32 = this.m32, m33 = this.m33,
167 			
168     determinant = this.determinant();
169 	
170 		if(determinant === 0) {
171 			throw("Matrix determinant is zero, can't invert.");
172 		}
173 	
174 		this.m11 = m22 * m33 - m23 * m32;
175 		this.m12 = m13 * m32 - m12 * m33;
176 		this.m13 = m12 * m23 - m13 * m22;
177 		this.m21 = m23 * m31 - m21 * m33;
178 		this.m22 = m11 * m33 - m13 * m31;
179 		this.m23 = m13 * m21 - m11 * m23;
180 		this.m31 = m21 * m32 - m22 * m31;
181 		this.m32 = m12 * m31 - m11 * m32;
182 		this.m33 = m11 * m22 - m12 * m21;
183 		
184 		this.scaleByScalar(1 / determinant);
185 		
186 		return this;
187 	},
188 	
189 	/**
190 	 * Copies the top left 3x3 matrix out of a Matrix4
191 	 * 
192 	 * @param {A3.Core.Math.Matrix4} matrix The Matrix4 to copy from
193 	 */
194 	copyMatrix4: function(matrix) {
195 		
196 		this.m11 = matrix.m11; this.m12 = matrix.m12; this.m13 = matrix.m13;
197 		this.m21 = matrix.m21; this.m22 = matrix.m22; this.m23 = matrix.m23;
198 		this.m31 = matrix.m31; this.m32 = matrix.m32; this.m33 = matrix.m33;
199 		
200 		return this;
201 	},
202 	
203 	/**
204 	 * Scales the whole matrix by a scalar value
205 	 * 
206 	 * @param {Number} scalar The value by which to multiply each matrix component
207 	 */
208 	scaleByScalar: function(scalar) {
209 		
210 		this.m11 *= scalar; this.m12 *= scalar; this.m13 *= scalar;
211 		this.m21 *= scalar; this.m22 *= scalar; this.m23 *= scalar;
212 		this.m31 *= scalar; this.m32 *= scalar; this.m33 *= scalar;
213 		
214 		return this;
215 		
216 	},
217 	
218 	/**
219 	 * Converts the Matrix3 to an array suitable for use in WebGL.
220 	 * 
221 	 * @param {Float32Array} holder The array to be populated with the matrix data
222 	 * @throws An error if a Float32Array is not provided
223 	 */
224 	toArray: function(holder) {
225 		
226 		// whinge if a holder isn't
227 		// provided to the function
228 		if(!holder) {
229 			throw "You must provide a Float32Array to populate";
230 		}
231 		
232 		/*
233 		 * Since this is used for sending data out to WebGL shaders,
234 		 * this populates using columns not rows.
235 		 */
236 		holder[0] = this.m11; holder[1] = this.m21; holder[2] = this.m31;
237 		holder[3] = this.m12; holder[4] = this.m22; holder[5] = this.m32;
238 		holder[6] = this.m13; holder[7] = this.m23; holder[8] = this.m33;
239 		
240 		return holder;
241 	}
242 };

334 	/**
335 	 * Creates a matrix to represent a frustum - 
336 	 * used for perspective and orthographic
337 	 * projections
338 	 *
339 	 * @see http://www.opengl.org/sdk/docs/man/xhtml/glFrustum.xml
340 	 *
341 	 * @param {Number} left The coordinate for the left clipping plane
342 	 * @param {Number} right The coordinate for the right clipping plane
343 	 * @param {Number} bottom The coordinate for the bottom clipping plane
344 	 * @param {Number} top The coordinate for the top clipping plane
345 	 * @param {Number} nearVal The distance to the near clipping plane
346 	 * @param {Number} farVal The distance to the far clipping plane
347 	 */
348 	frustum: function(left, right, bottom, top, nearVal, farVal) {
349 	
350 		var a = (right + left) / (right - left),
351         b = (top + bottom) / (top - bottom),
352         c = -(farVal + nearVal) / (farVal - nearVal),
353         d = -(2 * farVal * nearVal) / (farVal - nearVal),
354         x = (2 * nearVal) / (right - left),
355         y = (2 * nearVal) / (top - bottom);
356 			
357 		return this.set(
358 			x,  0,  a,  0,
359 			0,  y,  b,  0,
360 			0,  0,  c,  d,
361 			0,  0, -1,  0
362 		);
363 	},

365 	/**
366 	 * Creates a perspective matrix for a camera.
367 	 * Uses the frustrum to do the work of tying
368 	 * everything together
369 	 * 
370 	 * @param {Number} fieldOfView The field of view in degrees
371 	 * @param {Number} aspectRatio The ratio of width / height of the screen
372 	 * @param {Number} nearVal The distance to the near clipping plane
373 	 * @param {Number} farVal The distance to the far clipping plane
374 	 */
375 	perspective: function(fieldOfView, aspectRatio, nearVal, farVal) {
376 		
377 		var minX, maxX, minY, maxY;
378 		
379 		// This is the FOV / 2 converted to radians
380 		// and then multiplied by the near value - 
381 		// gives us our max y position at the front
382 		// of the frustum. The min is just the opposite
383 		maxY = nearVal * Math.tan(fieldOfView * Math.PI / 360);
384 		minY = -maxY;
385 		
386 		// the min and max x are then the same as
387 		// the min and max y multiplied by the
388 		// aspect ratio
389 		minX = minY * aspectRatio;
390 		maxX = maxY * aspectRatio;
391 		
392 		return this.frustum(
393 			minX, maxX, minY, maxY, nearVal, farVal
394 		);
395 		
396 	},

451 	/**
452 	 * Sets a matrix's rotation values in XYZ order
453 	 * based on the vector passed in.
454 	 * 
455 	 * @param {A3.Core.Math.Vector3} vector The vector specifying the rotation values in X,Y and Z
456 	 * 
457 	 * @see http://www.robertblum.com/articles/2005/02/14/decomposing-matrices [NB sign of
458 	 * sin values are reversed because we are working in a right handed system (WebGL)
459 	 * not a left handed one (DirectX)]
460 	 */
461 	setRotationFromVector: function(vector) {
462 		
463 		var x = vector.x, y = vector.y, z = vector.z,
464         cosX = Math.cos(x), sinX = Math.sin(x),
465         cosY = Math.cos(y), sinY = Math.sin(y),
466         cosZ = Math.cos(z), sinZ = Math.sin(z),
467 			
468         // some multiplications repeat so do them once
469         // and store the result
470         cosZsinX = cosZ * sinX,
471         sinXsinZ = sinX * sinZ,
472         cosXcosZ = cosX * cosZ;
473 			
474 		this.m11 = cosY * cosZ;
475 		this.m12 = cosY * -sinZ;
476 		this.m13 = sinY;
477 		this.m21 = cosZsinX * sinY + sinZ * cosX;
478 		this.m22 = sinXsinZ * -sinY + cosXcosZ;
479 		this.m23 = -sinX * cosY;
480 		this.m31 = cosXcosZ * -sinY + sinXsinZ;
481 		this.m32 = cosX * sinY * sinZ + cosZsinX;
482 		this.m33 = cosX * cosY;
483 		
484 		return this;
485 	},

487 	/**
488 	 * Multiplies this matrix by another
489 	 * 
490 	 * @param {A3.Core.Math.Matrix4} matrix The matrix by which to multiply the current one
491 	 */
492 	multiply: function(matrix) {
493 		
494 		// cache up all the values because we will change
495 		// them soon, and we need to refer to the old 
496 		// values during the calculations
497 		var m1_m11 = this.m11, m1_m12 = this.m12, m1_m13 = this.m13, m1_m14 = this.m14,
498         m1_m21 = this.m21, m1_m22 = this.m22, m1_m23 = this.m23, m1_m24 = this.m24,
499         m1_m31 = this.m31, m1_m32 = this.m32, m1_m33 = this.m33, m1_m34 = this.m34,
500         m1_m41 = this.m41, m1_m42 = this.m42, m1_m43 = this.m43, m1_m44 = this.m44,
501 			
502         m2_m11 = matrix.m11, m2_m12 = matrix.m12, m2_m13 = matrix.m13, m2_m14 = matrix.m14,
503         m2_m21 = matrix.m21, m2_m22 = matrix.m22, m2_m23 = matrix.m23, m2_m24 = matrix.m24,
504         m2_m31 = matrix.m31, m2_m32 = matrix.m32, m2_m33 = matrix.m33, m2_m34 = matrix.m34,
505         m2_m41 = matrix.m41, m2_m42 = matrix.m42, m2_m43 = matrix.m43, m2_m44 = matrix.m44;
506 		
507 		// row 1
508 		this.m11 = m1_m11 * m2_m11 + m1_m12 * m2_m21 + m1_m13 * m2_m31 + m1_m14 * m2_m41;
509 		this.m12 = m1_m11 * m2_m12 + m1_m12 * m2_m22 + m1_m13 * m2_m32 + m1_m14 * m2_m42;
510 		this.m13 = m1_m11 * m2_m13 + m1_m12 * m2_m23 + m1_m13 * m2_m33 + m1_m14 * m2_m43;
511 		this.m14 = m1_m11 * m2_m14 + m1_m12 * m2_m24 + m1_m13 * m2_m34 + m1_m14 * m2_m44;
512 		
513 		// row 2
514 		this.m21 = m1_m21 * m2_m11 + m1_m22 * m2_m21 + m1_m23 * m2_m31 + m1_m24 * m2_m41;
515 		this.m22 = m1_m21 * m2_m12 + m1_m22 * m2_m22 + m1_m23 * m2_m32 + m1_m24 * m2_m42;
516 		this.m23 = m1_m21 * m2_m13 + m1_m22 * m2_m23 + m1_m23 * m2_m33 + m1_m24 * m2_m43;
517 		this.m24 = m1_m21 * m2_m14 + m1_m22 * m2_m24 + m1_m23 * m2_m34 + m1_m24 * m2_m44;
518 		
519 		// row 3
520 		this.m31 = m1_m31 * m2_m11 + m1_m32 * m2_m21 + m1_m33 * m2_m31 + m1_m34 * m2_m41;
521 		this.m32 = m1_m31 * m2_m12 + m1_m32 * m2_m22 + m1_m33 * m2_m32 + m1_m34 * m2_m42;
522 		this.m33 = m1_m31 * m2_m13 + m1_m32 * m2_m23 + m1_m33 * m2_m33 + m1_m34 * m2_m43;
523 		this.m34 = m1_m31 * m2_m14 + m1_m32 * m2_m24 + m1_m33 * m2_m34 + m1_m34 * m2_m44;
524 		
525 		// row 4
526 		this.m41 = m1_m41 * m2_m11 + m1_m42 * m2_m21 + m1_m43 * m2_m31 + m1_m44 * m2_m41;
527 		this.m42 = m1_m41 * m2_m12 + m1_m42 * m2_m22 + m1_m43 * m2_m32 + m1_m44 * m2_m42;
528 		this.m43 = m1_m41 * m2_m13 + m1_m42 * m2_m23 + m1_m43 * m2_m33 + m1_m44 * m2_m43;
529 		this.m44 = m1_m41 * m2_m14 + m1_m42 * m2_m24 + m1_m43 * m2_m34 + m1_m44 * m2_m44;
530 		
531 		return this;
532 	},

534 	/**
535 	 * Calculates the determinant of the matrix. Primarily used for inverting the matrix
536 	 * 
537 	 * @see http://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/fourD/index.htm
538 	 */
539 	determinant: function() {
540 		
541 		return this.m14 * this.m23 * this.m32 * this.m41 - 
542            this.m13 * this.m24 * this.m32 * this.m41 - 
543            this.m14 * this.m22 * this.m33 * this.m41 +
544            this.m12 * this.m24 * this.m33 * this.m41 +
545 				
546            this.m13 * this.m22 * this.m34 * this.m41 -
547            this.m12 * this.m23 * this.m34 * this.m41 -
548            this.m14 * this.m23 * this.m31 * this.m42 +
549            this.m13 * this.m24 * this.m31 * this.m42 +
550 
551            this.m14 * this.m21 * this.m33 * this.m42 -
552            this.m11 * this.m24 * this.m33 * this.m42 - 
553            this.m13 * this.m21 * this.m34 * this.m42 +
554            this.m11 * this.m23 * this.m34 * this.m42 +
555 				
556            this.m14 * this.m22 * this.m31 * this.m43 -
557            this.m12 * this.m24 * this.m31 * this.m43 -
558            this.m14 * this.m21 * this.m32 * this.m43 +
559            this.m11 * this.m24 * this.m32 * this.m43 +
560 				
561            this.m12 * this.m21 * this.m34 * this.m43 -
562            this.m11 * this.m22 * this.m34 * this.m43 -
563            this.m13 * this.m22 * this.m31 * this.m44 +
564            this.m12 * this.m23 * this.m31 * this.m44 +
565 				
566            this.m13 * this.m21 * this.m32 * this.m44 -
567            this.m11 * this.m23 * this.m32 * this.m44 -
568            this.m12 * this.m21 * this.m33 * this.m44 +
569            this.m11 * this.m22 * this.m33 * this.m44;

572 	/**
573 	 * Inverts the matrix.
574 	 * 
575 	 * @throws An error if the matrix determinant is zero
576    * @see http://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/fourD/index.htm
577 	 */
578 	invert: function() {
579 		
580 		var m11 = this.m11, m12 = this.m12, m13 = this.m13, m14 = this.m14,
581         m21 = this.m21, m22 = this.m22, m23 = this.m23, m24 = this.m24,
582         m31 = this.m31, m32 = this.m32, m33 = this.m33, m34 = this.m34,
583         m41 = this.m41, m42 = this.m42, m43 = this.m43, m44 = this.m44,
584 			
585         determinant = this.determinant();
586 	
587 		if(determinant === 0) {
588 			throw("Matrix determinant is zero, can't invert.");
589 		}
590 	
591 		// check this out... like, a *lot* of calculations
592 		this.m11 = m23 * m34 * m42 - m24 * m33 * m42 + m24 * m32 * m43 - m22 * m34 * m43 - m23 * m32 * m44 + m22 * m33 * m44;
593 		this.m12 = m14 * m33 * m42 - m13 * m34 * m42 - m14 * m32 * m43 + m12 * m34 * m43 + m13 * m32 * m44 - m12 * m33 * m44;
594 		this.m13 = m13 * m24 * m42 - m14 * m23 * m42 + m14 * m22 * m43 - m12 * m24 * m43 - m13 * m22 * m44 + m12 * m23 * m44;
595 		this.m14 = m14 * m23 * m32 - m13 * m24 * m32 - m14 * m22 * m33 + m12 * m24 * m33 + m13 * m22 * m34 - m12 * m23 * m34;
596 		this.m21 = m24 * m33 * m41 - m23 * m34 * m41 - m24 * m31 * m43 + m21 * m34 * m43 + m23 * m31 * m44 - m21 * m33 * m44;
597 		this.m22 = m13 * m34 * m41 - m14 * m33 * m41 + m14 * m31 * m43 - m11 * m34 * m43 - m13 * m31 * m44 + m11 * m33 * m44;
598 		this.m23 = m14 * m23 * m41 - m13 * m24 * m41 - m14 * m21 * m43 + m11 * m24 * m43 + m13 * m21 * m44 - m11 * m23 * m44;
599 		this.m24 = m13 * m24 * m31 - m14 * m23 * m31 + m14 * m21 * m33 - m11 * m24 * m33 - m13 * m21 * m34 + m11 * m23 * m34;
600 		this.m31 = m22 * m34 * m41 - m24 * m32 * m41 + m24 * m31 * m42 - m21 * m34 * m42 - m22 * m31 * m44 + m21 * m32 * m44;
601 		this.m32 = m14 * m32 * m41 - m12 * m34 * m41 - m14 * m31 * m42 + m11 * m34 * m42 + m12 * m31 * m44 - m11 * m32 * m44;
602 		this.m33 = m12 * m24 * m41 - m14 * m22 * m41 + m14 * m21 * m42 - m11 * m24 * m42 - m12 * m21 * m44 + m11 * m22 * m44;
603 		this.m34 = m14 * m22 * m31 - m12 * m24 * m31 - m14 * m21 * m32 + m11 * m24 * m32 + m12 * m21 * m32 - m11 * m22 * m34;
604 		this.m41 = m23 * m32 * m41 - m22 * m33 * m41 - m23 * m31 * m42 + m21 * m33 * m42 + m22 * m31 * m43 - m21 * m32 * m43;
605 		this.m42 = m12 * m33 * m41 - m13 * m32 * m41 + m13 * m31 * m42 - m11 * m33 * m42 - m12 * m31 * m43 + m11 * m32 * m43;
606 		this.m43 = m13 * m22 * m41 - m12 * m23 * m41 - m13 * m21 * m42 + m11 * m23 * m42 + m12 * m21 * m43 - m11 * m22 * m43;
607 		this.m44 = m12 * m23 * m31 - m13 * m22 * m31 + m13 * m21 * m32 - m11 * m23 * m32 - m12 * m21 * m33 + m11 * m22 * m33;
608 		
609 		this.scaleByScalar(1 / determinant);
610 		
611 		return this;
612 		
613 	},

615 	/**
616 	 * Modifies the matrix to look at a specific place
617 	 * 
618 	 * @param {A3.Core.Math.Vector3} eye The position of the eye. Typically the camera or object position
619 	 * @param {A3.Core.Math.Vector3} target The position in 3D space to look at
620 	 * @param {A3.Core.Math.Vector3} up The world's up vector
621 	 */
622 	lookAt: function(eye, target, up) {
623 		
624 		var dirVec    = new A3.Core.Math.Vector3(),
625         rightVec  = new A3.Core.Math.Vector3(),
626         upVec     = new A3.Core.Math.Vector3();
627 		
628 		dirVec
629 			.copy(eye)
630 			.subtract(target)
631 			.normalize();
632 			
633 		// ensure we have a valid direction
634 		if(dirVec.length() === 0) {
635       dirVec.z = 1;
636 		}
637 		
638 		// get the right vector
639 		rightVec
640 			.copy(up)
641 			.cross(dirVec)
642 			.normalize();
643 		
644 		// also check that the right vector is ok
645 		if(rightVec.length() === 0) {
646 		  
647 			dirVec.z += 0.1;
648 			
649 			// redo the right vector calculation
650 			rightVec
651 				.copy(up)
652 				.cross(dirVec)
653 				.normalize();
654 		}
655 		
656 		// and now we have the dir and right, we can
657 		// work out what the real up vector should be
658 		upVec
659 			.copy(dirVec)
660 			.cross(rightVec)
661 			.normalize();
662 			
663 		// now we set the matrix values
664 		// based on these vectors
665 		this.m11 = rightVec.x; this.m12 = upVec.x; this.m13 = dirVec.x;
666 		this.m21 = rightVec.y; this.m22 = upVec.y; this.m23 = dirVec.y;
667 		this.m31 = rightVec.z; this.m32 = upVec.z; this.m33 = dirVec.z;
668 		
669 		return this;
670 	},

691 		 * should occupy the 12th, 13th and 14th slot in the array which if you check
692 		 * means that this is going out in colums.
693 		 */
694 		holder[0]  = this.m11; holder[1]  = this.m21; holder[2]  = this.m31;  holder[3]  = this.m41;
695 		holder[4]  = this.m12; holder[5]  = this.m22; holder[6]  = this.m32;  holder[7]  = this.m42;
696 		holder[8]  = this.m13; holder[9]  = this.m23; holder[10] = this.m33;  holder[11] = this.m43;
697 		holder[12] = this.m14; holder[13] = this.m24; holder[14] = this.m34;  holder[15] = this.m44;
698

700 	},

1343 			// another object we should remove
1344 			// it first

1363 	 * 
1364 	 * @param {Object3D} child The child object to be removed
1365 	 */

1367 		var position = this.children.indexOf(child);
1368 		
1369 		// ooh, use of tilde
1370 		if(~position) {
1371 			this.children.splice(position, 1);
1372 			this.dirty = true;
1373 		}
1374 	},
1375 	
1376 	/**
1377 	 * Updates an object's internal matrix.
1378 	 * 
1379 	 * @param {Matrix4} parentMatrixWorld The world matrix of the parent object
1380 	 * @param {Boolean} parentIsDirty Whether or not the parent's dirty flag is set
1381 	 */
1382 	update: function(parentMatrixWorld, parentIsDirty) {
1383 		
1384 		var childCount		= this.children.length;
1385 		
1386 		// if the dirty value has already been
1387 		// set, the parent is dirty or any of
1388 		// the vectors have changed this is dirty
1389 		this.dirty		= parentIsDirty           ||
1390 						        this.dirty              || 
1391 						        this.position.isDirty() || 
1392 						        this.rotation.isDirty() || 
1393 						        this.scale.isDirty();
1394 						
1395 		// if it has changed in any way we
1396 		// need to update the object's matrix
1397 		if(this.dirty) {
1398 			
1399 			// reset it then we update it with the
1400 			// specifics of our object
1401 			this.matrix.identity();
1402 			this.matrix.setTranslationFromVector(this.position);
1403 			
1404 			if(!this.target) {
1405 				this.matrix.setRotationFromVector(this.rotation);
1406 			} else {
1407 				// TODO This assumes they're both in the same coordinate space - need to fix
1408 				this.matrix.lookAt(this.position, this.target, this.up);	
1409 			}
1410 			
1411 			this.matrix.scaleByVector(this.scale);
1412 		
1413 			/**
1414 			 * if no parent world matrix
1415 			 * exists, i.e. this is the root
1416 			 * then create one and set it to the
1417 			 * identity matrix
1418 			 */
1419 			if(!parentMatrixWorld) {
1420 				this.matrixWorld.copy(this.matrix);
1421 			} else {
1422 				// copies the parent's world matrix
1423 				this.matrixWorld.copy(parentMatrixWorld);
1424 				
1425 				// multiply the world matrix with the object matrix
1426 				this.matrixWorld.multiply(this.matrix);
1427 			}
1428 			
1429 			/**
1430 			 * If this Object3D has a normals matrix
1431 			 * then we calculate that here
1432 			 * 
1433 			 * @see http://www.lighthouse3d.com/tutorials/glsl-tutorial/the-normal-matrix/
1434 			 */
1435 			if(!!this.matrixNormals) {
1436 				this.matrixNormals
1437 					.copyMatrix4(this.matrixWorld)
1438 					.invert()
1439 					.transpose();
1440 			}
1441 		
1442 			/**
1443 			 * If this Object3D has an inverse matrix
1444 			 * property, like the camera does, calculate it
1445 			 */
1446 			if(!!this.inverseMatrix) {
1447 				// start with the world matrix
1448 				// and then invert it
1449 				this.inverseMatrix.copy(this.matrixWorld);
1450 				this.inverseMatrix.invert();
1451 			}
1452 		}
1453 			
1454 		// pass through to the children
1455 		while(childCount--) {
1456 			this.children[childCount].update(this.matrixWorld, this.dirty);
1457 		}
1458 		
1459 		// reset the dirty values to false
1460 		this.position.resetDirty();
1461 		this.rotation.resetDirty();
1462 		this.scale.resetDirty();
1463 		this.dirty = false;
1464 	},
