152 	/**
153 	 * Creates a matrix to represent a frustum - 
154 	 * used for perspective and orthographic
155 	 * projections
156 	 *
157 	 * @see http://www.opengl.org/sdk/docs/man/xhtml/glFrustum.xml
158 	 *
159 	 * @param {Number} left The coordinate for the left clipping plane
160 	 * @param {Number} right The coordinate for the right clipping plane
161 	 * @param {Number} bottom The coordinate for the bottom clipping plane
162 	 * @param {Number} top The coordinate for the top clipping plane
163 	 * @param {Number} nearVal The distance to the near clipping plane
164 	 * @param {Number} farVal The distance to the far clipping plane
165 	 */
166 	frustum: function(left, right, bottom, top, nearVal, farVal) {
167 	
168 		var a = (right + left) / (right - left),
169 			b = (top + bottom) / (top - bottom),
170 			c = -(farVal + nearVal) / (farVal - nearVal),
171 			d = -(2 * farVal * nearVal) / (farVal - nearVal),
172 			x = (2 * nearVal) / (right - left),
173 			y = (2 * nearVal) / (top - bottom);
174 			
175 		this.set(
176 			x,  0,  a,  0,
177 			0,  y,  b,  0,
178 			0,  0,  c,  d,
179 			0,  0, -1,  0
180 		);
181 	},

183 	/**
184 	 * Creates a perspective matrix for a camera.
185 	 * Uses the frustrum to do the work of tying
186 	 * everything together
187 	 * 
188 	 * @param {Number} fieldOfView The field of view in degrees
189 	 * @param {Number} aspectRatio The ratio of width / height of the screen
190 	 * @param {Number} nearVal The distance to the near clipping plane
191 	 * @param {Number} farVal The distance to the far clipping plane
192 	 */
193 	perspective: function(fieldOfView, aspectRatio, nearVal, farVal) {
194 		
195 		var minX, maxX, minY, maxY;
196 		
197 		// This is the FOV / 2 converted to radians
198 		// and then multiplied by the near value - 
199 		// gives us our max y position at the front
200 		// of the frustum. The min is just the opposite
201 		maxY = nearVal * Math.tan(fieldOfView * Math.PI / 360);
202 		minY = -maxY;
203 		
204 		// the min and max x are then the same as
205 		// the min and max y multiplied by the
206 		// aspect ratio
207 		minX = minY * aspectRatio;
208 		maxX = maxY * aspectRatio;
209 		
210 		return this.frustum(
211 			minX, maxX, minY, maxY, nearVal, farVal
212 		);
213 		
214 	},

269 	/**
270 	 * Sets a matrix's rotation values in XYZ order
271 	 * based on the vector passed in.
272 	 * 
273 	 * @param {A3.Core.Math.Vector3} vector The vector specifying the rotation values in X,Y and Z
274 	 * 
275 	 * @see http://www.robertblum.com/articles/2005/02/14/decomposing-matrices [NB sign of
276 	 * sin values are reversed because we are working in a right handed system (WebGL)
277 	 * not a left handed one (DirectX)]
278 	 */
279 	setRotationFromVector: function(vector) {
280 		
281 		var x = vector.x, y = vector.y, z = vector.z,
282 			cosX = Math.cos(x), sinX = Math.sin(x),
283 			cosY = Math.cos(y), sinY = Math.sin(y),
284 			cosZ = Math.cos(z), sinZ = Math.sin(z),
285 			
286 			// some multiplications repeat so do them once
287 			// and store the result
288 			cosZsinX = cosZ * sinX,
289 			sinXsinZ = sinX * sinZ,
290 			cosXcosZ = cosX * cosZ;
291 			
292 		this.m11 = cosY * cosZ;
293 		this.m12 = cosY * -sinZ;
294 		this.m13 = sinY;
295 		this.m21 = cosZsinX * sinY + sinZ * cosX;
296 		this.m22 = sinXsinZ * -sinY + cosXcosZ;
297 		this.m23 = -sinX * cosY;
298 		this.m31 = cosXcosZ * -sinY + sinXsinZ;
299 		this.m32 = cosX * sinY * sinZ + cosZsinX;
300 		this.m33 = cosX * cosY;
301 		
302 		return this;
303 	},

305 	/**
306 	 * Multiplies this matrix by another
307 	 * 
308 	 * @param {A3.Core.Math.Matrix4} matrix The matrix by which to multiply the current one
309 	 */
310 	multiply: function(matrix) {
311 		
312 		// cache up all the values because we will change
313 		// them soon, and we need to refer to the old 
314 		// values during the calculations
315 		var m1_m11 = this.m11, m1_m12 = this.m12, m1_m13 = this.m13, m1_m14 = this.m14,
316 			m1_m21 = this.m21, m1_m22 = this.m22, m1_m23 = this.m23, m1_m24 = this.m24,
317 			m1_m31 = this.m31, m1_m32 = this.m32, m1_m33 = this.m33, m1_m34 = this.m34,
318 			m1_m41 = this.m41, m1_m42 = this.m42, m1_m43 = this.m43, m1_m44 = this.m44,
319 			
320 			m2_m11 = matrix.m11, m2_m12 = matrix.m12, m2_m13 = matrix.m13, m2_m14 = matrix.m14,
321 			m2_m21 = matrix.m21, m2_m22 = matrix.m22, m2_m23 = matrix.m23, m2_m24 = matrix.m24,
322 			m2_m31 = matrix.m31, m2_m32 = matrix.m32, m2_m33 = matrix.m33, m2_m34 = matrix.m34,
323 			m2_m41 = matrix.m41, m2_m42 = matrix.m42, m2_m43 = matrix.m43, m2_m44 = matrix.m44;
324 		
325 		// row 1
326 		this.m11 = m1_m11 * m2_m11 + m1_m12 * m2_m21 + m1_m13 * m2_m31 + m1_m14 * m2_m41;
327 		this.m12 = m1_m11 * m2_m12 + m1_m12 * m2_m22 + m1_m13 * m2_m32 + m1_m14 * m2_m42;
328 		this.m13 = m1_m11 * m2_m13 + m1_m12 * m2_m23 + m1_m13 * m2_m33 + m1_m14 * m2_m43;
329 		this.m14 = m1_m11 * m2_m14 + m1_m12 * m2_m24 + m1_m13 * m2_m34 + m1_m14 * m2_m44;
330 		
331 		// row 2
332 		this.m21 = m1_m21 * m2_m11 + m1_m22 * m2_m21 + m1_m23 * m2_m31 + m1_m24 * m2_m41;
333 		this.m22 = m1_m21 * m2_m12 + m1_m22 * m2_m22 + m1_m23 * m2_m32 + m1_m24 * m2_m42;
334 		this.m23 = m1_m21 * m2_m13 + m1_m22 * m2_m23 + m1_m23 * m2_m33 + m1_m24 * m2_m43;
335 		this.m24 = m1_m21 * m2_m14 + m1_m22 * m2_m24 + m1_m23 * m2_m34 + m1_m24 * m2_m44;
336 		
337 		// row 3
338 		this.m31 = m1_m31 * m2_m11 + m1_m32 * m2_m21 + m1_m33 * m2_m31 + m1_m34 * m2_m41;
339 		this.m32 = m1_m31 * m2_m12 + m1_m32 * m2_m22 + m1_m33 * m2_m32 + m1_m34 * m2_m42;
340 		this.m33 = m1_m31 * m2_m13 + m1_m32 * m2_m23 + m1_m33 * m2_m33 + m1_m34 * m2_m43;
341 		this.m34 = m1_m31 * m2_m14 + m1_m32 * m2_m24 + m1_m33 * m2_m34 + m1_m34 * m2_m44;
342 		
343 		// row 4
344 		this.m41 = m1_m41 * m2_m11 + m1_m42 * m2_m21 + m1_m43 * m2_m31 + m1_m44 * m2_m41;
345 		this.m42 = m1_m41 * m2_m12 + m1_m42 * m2_m22 + m1_m43 * m2_m32 + m1_m44 * m2_m42;
346 		this.m43 = m1_m41 * m2_m13 + m1_m42 * m2_m23 + m1_m43 * m2_m33 + m1_m44 * m2_m43;
347 		this.m44 = m1_m41 * m2_m14 + m1_m42 * m2_m24 + m1_m43 * m2_m34 + m1_m44 * m2_m44;
348 		
349 		return this;
350 	},

352 	/**
353 	 * Calculates the determinant of the matrix. Used for inverting the matrix primarily
354 	 * 
355 	 * @see http://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/fourD/index.htm
356 	 */
357 	determinant: function() {
358 		
359 		return this.m14 * this.m23 * this.m32 * this.m41 - 
360 				this.m13 * this.m24 * this.m32 * this.m41 - 
361 				this.m14 * this.m22 * this.m33 * this.m41 +
362 				this.m12 * this.m24 * this.m33 * this.m41 +
363 				
364 				this.m13 * this.m22 * this.m34 * this.m41 -
365 				this.m12 * this.m23 * this.m34 * this.m41 -
366 				this.m14 * this.m23 * this.m31 * this.m42 +
367 				this.m13 * this.m24 * this.m31 * this.m42 +
368 
369 				this.m14 * this.m21 * this.m33 * this.m42 -
370 				this.m11 * this.m24 * this.m33 * this.m42 - 
371 				this.m13 * this.m21 * this.m34 * this.m42 +
372 				this.m11 * this.m23 * this.m34 * this.m42 +
373 				
374 				this.m14 * this.m22 * this.m31 * this.m43 -
375 				this.m12 * this.m24 * this.m31 * this.m43 -
376 				this.m14 * this.m21 * this.m32 * this.m43 +
377 				this.m11 * this.m24 * this.m32 * this.m43 +
378 				
379 				this.m12 * this.m21 * this.m34 * this.m43 -
380 				this.m11 * this.m22 * this.m34 * this.m43 -
381 				this.m13 * this.m22 * this.m31 * this.m44 +
382 				this.m12 * this.m23 * this.m31 * this.m44 +
383 				
384 				this.m13 * this.m21 * this.m32 * this.m44 -
385 				this.m11 * this.m23 * this.m32 * this.m44 -
386 				this.m12 * this.m21 * this.m33 * this.m44 +
387 				this.m11 * this.m22 * this.m33 * this.m44;
388 	},

390 	/**
391 	 * Inverts a matrix.
392 	 * @throws An error if the matrix determinant is zero
393 	 */
394 	invert: function() {
395 		
396 		var m11 = this.m11, m12 = this.m12, m13 = this.m13, m14 = this.m14,
397 			m21 = this.m21, m22 = this.m22, m23 = this.m23, m24 = this.m24,
398 			m31 = this.m31, m32 = this.m32, m33 = this.m33, m34 = this.m34,
399 			m41 = this.m41, m42 = this.m42, m43 = this.m43, m44 = this.m44,
400 			
401 			determinant = this.determinant();
402 	
403 		if(determinant === 0) {
404 			throw("Matrix determinant is zero, can't invert.");
405 		}
406 	
407 		this.m11 = m23 * m34 * m42 - m24 * m33 * m42 + m24 * m32 * m43 - m22 * m34 * m43 - m23 * m32 * m44 + m22 * m33 * m44;
408 		this.m12 = m14 * m33 * m42 - m13 * m34 * m42 - m14 * m32 * m43 + m12 * m34 * m43 + m13 * m32 * m44 - m12 * m33 * m44;
409 		this.m13 = m13 * m24 * m42 - m14 * m23 * m42 + m14 * m22 * m43 - m12 * m24 * m43 - m13 * m22 * m44 + m12 * m23 * m44;
410 		this.m14 = m14 * m23 * m32 - m13 * m24 * m32 - m14 * m22 * m33 + m12 * m24 * m33 + m13 * m22 * m34 - m12 * m23 * m34;
411 		this.m21 = m24 * m33 * m41 - m23 * m34 * m41 - m24 * m31 * m43 + m21 * m34 * m43 + m23 * m31 * m44 - m21 * m33 * m44;
412 		this.m22 = m13 * m34 * m41 - m14 * m33 * m41 + m14 * m31 * m43 - m11 * m34 * m43 - m13 * m31 * m44 + m11 * m33 * m44;
413 		this.m23 = m14 * m23 * m41 - m13 * m24 * m41 - m14 * m21 * m43 + m11 * m24 * m43 + m13 * m21 * m44 - m11 * m23 * m44;
414 		this.m24 = m13 * m24 * m31 - m14 * m23 * m31 + m14 * m21 * m33 - m11 * m24 * m33 - m13 * m21 * m34 + m11 * m23 * m34;
415 		this.m31 = m22 * m34 * m41 - m24 * m32 * m41 + m24 * m31 * m42 - m21 * m34 * m42 - m22 * m31 * m44 + m21 * m32 * m44;
416 		this.m32 = m14 * m32 * m41 - m12 * m34 * m41 - m14 * m31 * m42 + m11 * m34 * m42 + m12 * m31 * m44 - m11 * m32 * m44;
417 		this.m33 = m12 * m24 * m41 - m14 * m22 * m41 + m14 * m21 * m42 - m11 * m24 * m42 - m12 * m21 * m44 + m11 * m22 * m44;
418 		this.m34 = m14 * m22 * m31 - m12 * m24 * m31 - m14 * m21 * m32 + m11 * m24 * m32 + m12 * m21 * m32 - m11 * m22 * m34;
419 		this.m41 = m23 * m32 * m41 - m22 * m33 * m41 - m23 * m31 * m42 + m21 * m33 * m42 + m22 * m31 * m43 - m21 * m32 * m43;
420 		this.m42 = m12 * m33 * m41 - m13 * m32 * m41 + m13 * m31 * m42 - m11 * m33 * m42 - m12 * m31 * m43 + m11 * m32 * m43;
421 		this.m43 = m13 * m22 * m41 - m12 * m23 * m41 - m13 * m21 * m42 + m11 * m23 * m42 + m12 * m21 * m43 - m11 * m22 * m43;
422 		this.m44 = m12 * m23 * m31 - m13 * m22 * m31 + m13 * m21 * m32 - m11 * m23 * m32 - m12 * m21 * m33 + m11 * m22 * m33;
423 		
424 		this.scaleByScalar(1/determinant);
425 		
426 		return this;
427

430 	/**
431 	 * Modifies the matrix to look at a specific place
432 	 * 
433 	 * @param {A3.Core.Math.Vector3} eye The position of the eye. Typically the camera or object position
434 	 * @param {A3.Core.Math.Vector3} target The position in 3D space to look at
435 	 * @param {A3.Core.Math.Vector3} up The world's up vector
436 	 */
437 	lookAt: function(eye, target, up) {
438 		
439 		var dirVec		= eye.subtract(target, true).normalize(),
440 			rightVec	= null,
441 			upVec		= null;
442 			
443 		// ensure we have a valid direction
444 		if(dirVec.length() === 0) {
445 			dirVec.z = 1;
446 		}
447 		
448 		// get the right vector
449 		rightVec = up.cross(dirVec, true).normalize();
450 		
451 		// also check that the right vector is ok
452 		if(rightVec.length() === 0) {
453 			dirVec.z += 0.001;
454 			
455 			// redo the right vector calculation
456 			rightVec = up.cross(dirVec, true).normalize();
457 		}
458 		
459 		// and now we have the dir and right, we can
460 		// work out what the real up vector should be
461 		upVec = dirVec.cross(rightVec, true).normalize();
462 		
463 		// now we set the matrix values
464 		// based on these vectors
465 		this.m11 = rightVec.x; this.m12 = upVec.x; this.m13 = dirVec.x;
466 		this.m21 = rightVec.y; this.m22 = upVec.y; this.m23 = dirVec.y;
467 		this.m31 = rightVec.z; this.m32 = upVec.z; this.m33 = dirVec.z;
468 		
469 		return this;
470 	},

472 	/**
473 	 * Converts the Matrix4 to an array suitable for use in WebGL.
474 	 * 
475 	 * @param {Float32Array[]} holder The array to be populated with the matrix data
476 	 */
477 	toArray: function(holder) {
478 		
479 		// create a holder if one isn't
480 		// provided to the function

917 	/**
918 	 * Updates an object's internal matrix.
919 	 * 
920 	 * @param {Matrix4} parentMatrixWorld The world matrix of the parent object
921 	 * @param {Boolean} parentIsDirty Whether or not the parent's dirty flag is set
922 	 */
923 	update: function(parentMatrixWorld, parentIsDirty) {
924 		
925 		var childCount = this.children.length;
926 		
927 		// if the dirty value has already been
928 		// set, the parent is dirty or any of
929 		// the vectors have changed this is dirty
930 		this.dirty		= parentIsDirty			||
931 						this.dirty				|| 
932 						this.position.isDirty() || 
933 						this.rotation.isDirty() || 
934 						this.scale.isDirty();
935 						
936 		// if it has changed in any way we
937 		// need to update the object's matrix
938 		if(this.dirty) {
939 			
940 			// reset it then we update it with the
941 			// specifics of our object
942 			this.matrix.identity();
943 			this.matrix.setTranslationFromVector(this.position);
944 			
945 			if(!this.target) {
946 				this.matrix.setRotationFromVector(this.rotation);
947 			} else {
948 				this.matrix.lookAt(this.position, this.target, this.up);	
949 			}
950 			
951 			this.matrix.scaleByVector(this.scale);
952 		
953 			/**
954 			 * if no parent world matrix
955 			 * exists, i.e. this is the root
956 			 * then create one and set it to the
957 			 * identity matrix
958 			 */
959 			if(!parentMatrixWorld) {
960 				this.matrixWorld.clone(this.matrix);
961 			} else {
962 				// clone the parent's world matrix
963 				this.matrixWorld.clone(parentMatrixWorld);
964 				
965 				// multiply the world matrix with the object matrix
966 				this.matrixWorld.multiply(this.matrix);
967 			}
968 		
969 			/**
970 			 * If this Object3D has an inverse matrix
971 			 * property, like the camera does, calculate it
972 			 */
973 			if(!!this.inverseMatrix) {
974 				// start with the world matrix
975 				// and then invert it
976 				this.inverseMatrix.clone(this.matrixWorld);
977 				this.inverseMatrix.invert();
978 			}
979 		}
980 			
981 		// pass through to the children
982 		while(childCount--) {
983 			this.children[childCount].update(this.matrixWorld, this.dirty);
984 		}
985 	},
